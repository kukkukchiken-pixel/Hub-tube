    <script>
        let videos = [];
        let currentFilter = 'all';
        let selectedFile = null;
        
        // Load videos from storage
        function loadVideos() {
            const stored = localStorage.getItem('hubtube_videos');
            if (stored) {
                videos = JSON.parse(stored);
            }
            renderVideos();
        }
        
        // Save videos to storage
        function saveVideos() {
            localStorage.setItem('hubtube_videos', JSON.stringify(videos));
        }
        
        // Render videos
        function renderVideos() {
            const grid = document.getElementById('videoGrid');
            let filteredVideos = [...videos];
            
            // Apply filters
            if (currentFilter === 'random') {
                filteredVideos = shuffleArray(filteredVideos);
                document.getElementById('sectionTitle').textContent = 'Random Videos';
            } else if (currentFilter === 'latest') {
                filteredVideos.sort((a, b) => b.timestamp - a.timestamp);
                document.getElementById('sectionTitle').textContent = 'Latest Videos';
            } else if (currentFilter === 'oldest') {
                filteredVideos.sort((a, b) => a.timestamp - b.timestamp);
                document.getElementById('sectionTitle').textContent = 'Oldest Videos';
            } else if (currentFilter === 'popular') {
                filteredVideos.sort((a, b) => b.views - a.views);
                document.getElementById('sectionTitle').textContent = 'Most Viewed';
            } else {
                document.getElementById('sectionTitle').textContent = 'All Videos';
            }
            
            if (filteredVideos.length === 0) {
                grid.innerHTML = `
                    <div class="empty-state">
                        <div class="empty-icon">üìπ</div>
                        <h3>No videos yet</h3>
                        <p>Upload your first video to get started!</p>
                    </div>
                `;
                return;
            }
            
            grid.innerHTML = filteredVideos.map((video, index) => `
                <div class="video-card" onclick="playVideo(${videos.indexOf(video)})">
                    <div class="video-thumbnail">
                        ${video.type === 'video' ? 
                            `<video src="${video.file}" preload="metadata"></video>` :
                            `<img src="${video.file}" alt="${video.title}">`
                        }
                        <div class="play-icon">‚ñ∂Ô∏è</div>
                        ${video.duration ? `<div class="video-duration">${video.duration}</div>` : ''}
                    </div>
                    <div class="video-info">
                        <div class="video-title">${video.title}</div>
                        <div class="video-meta">
                            <div class="video-stats">
                                <span>üëÅÔ∏è ${formatViews(video.views)} views</span>
                                <span>‚Ä¢</span>
                                <span>${formatDate(video.timestamp)}</span>
                            </div>
                        </div>
                    </div>
                </div>
            `).join('');
        }
        
        // Filter videos
        function filterVideos(filter) {
            currentFilter = filter;
            
            // Update active tab
            document.querySelectorAll('.tab').forEach(tab => {
                tab.classList.remove('active');
            });
            event.target.classList.add('active');
            
            renderVideos();
        }
        
        // Search videos
        function searchVideos() {
            const query = document.getElementById('searchInput').value.toLowerCase();
            if (!query) {
                renderVideos();
                return;
            }
            
            const grid = document.getElementById('videoGrid');
            const results = videos.filter(v => 
                v.title.toLowerCase().includes(query) || 
                v.description.toLowerCase().includes(query)
            );
            
            document.getElementById('sectionTitle').textContent = `Search results for "${query}"`;
            
            if (results.length === 0) {
                grid.innerHTML = `
                    <div class="empty-state">
                        <div class="empty-icon">üîç</div>
                        <h3>No results found</h3>
                        <p>Try different keywords</p>
                    </div>
                `;
                return;
            }
            
            grid.innerHTML = results.map((video, index) => `
                <div class="video-card" onclick="playVideo(${videos.indexOf(video)})">
                    <div class="video-thumbnail">
                        ${video.type === 'video' ? 
                            `<video src="${video.file}" preload="metadata"></video>` :
                            `<img src="${video.file}" alt="${video.title}">`
                        }
                        <div class="play-icon">‚ñ∂Ô∏è</div>
                        ${video.duration ? `<div class="video-duration">${video.duration}</div>` : ''}
                    </div>
                    <div class="video-info">
                        <div class="video-title">${video.title}</div>
                        <div class="video-meta">
                            <div class="video-stats">
                                <span>üëÅÔ∏è ${formatViews(video.views)} views</span>
                                <span>‚Ä¢</span>
                                <span>${formatDate(video.timestamp)}</span>
                            </div>
                        </div>
                    </div>
                </div>
            `).join('');
        }
        
        function handleSearchKeypress(e) {
            if (e.key === 'Enter') searchVideos();
        }
        
        // Upload Modal
        function openUploadModal() {
            document.getElementById('uploadModal').classList.add('show');
        }
        
        function closeUploadModal() {
            document.getElementById('uploadModal').classList.remove('show');
            resetUploadForm();
        }
        
        function handleFileSelect(e) {
            const file = e.target.files[0];
            if (!file) return;
            
            selectedFile = file;
            const reader = new FileReader();
            
            reader.onload = function(event) {
                const previewContainer = document.getElementById('previewContainer');
                const videoPreview = document.getElementById('videoPreview');
                const imagePreview = document.getElementById('imagePreview');
                
                previewContainer.classList.add('show');
                
                if (file.type.startsWith('video/')) {
                    videoPreview.src = event.target.result;
                    videoPreview.style.display = 'block';
                    imagePreview.style.display = 'none';
                } else {
                    imagePreview.src = event.target.result;
                    imagePreview.style.display = 'block';
                    videoPreview.style.display = 'none';
                }
                
                document.getElementById('uploadSubmitBtn').disabled = false;
            };
            
            reader.readAsDataURL(file);
        }
        
        function uploadVideo() {
            const title = document.getElementById('videoTitle').value.trim();
            const description = document.getElementById('videoDescription').value.trim();
            
            if (!title) {
                alert('Please enter a title');
                return;
            }
            
            if (!selectedFile) {
                alert('Please select a file');
                return;
            }
            
            const reader = new FileReader();
            reader.onload = function(e) {
                const video = {
                    id: Date.now(),
                    title: title,
                    description: description,
                    file: e.target.result,
                    type: selectedFile.type.startsWith('video/') ? 'video' : 'image',
                    views: 0,
                    timestamp: Date.now(),
                    duration: null
                };
                
                videos.unshift(video);
                saveVideos();
                renderVideos();
                closeUploadModal();
                
                alert('‚úÖ Upload successful!');
            };
            
            reader.readAsDataURL(selectedFile);
        }
        
        function resetUploadForm() {
            document.getElementById('videoTitle').value = '';
            document.getElementById('videoDescription').value = '';
            document.getElementById('fileInput').value = '';
            document.getElementById('previewContainer').classList.remove('show');
            document.getElementById('uploadSubmitBtn').disabled = true;
            selectedFile = null;
        }
        
        // Play video
        function playVideo(index) {
            const video = videos[index];
            video.views++;
            saveVideos();
            
            const modal = document.getElementById('playerModal');
            const container = document.getElementById('playerContainer');
            
            if (video.type === 'video') {
                container.innerHTML = `<video src="${video.file}" controls autoplay style="max-width: 100%; max-height: 70vh;"></video>`;
            } else {
                container.innerHTML = `<img src="${video.file}" style="max-width: 100%; max-height: 70vh;">`;
            }
            
            document.getElementById('playerTitle').textContent = video.title;
            document.getElementById('playerViews').textContent = `üëÅÔ∏è ${formatViews(video.views)} views`;
            document.getElementById('playerDate').textContent = `üìÖ ${formatDate(video.timestamp)}`;
            document.getElementById('playerDescription').textContent = video.description || 'No description';
            
            modal.classList.add('show');
            renderVideos();
        }
        
        function closePlayer() {
            const modal = document.getElementById('playerModal');
            modal.classList.remove('show');
            document.getElementById('playerContainer').innerHTML = '';
        }
        
        // Helper functions
        function formatViews(views) {
            if (views >= 1000000) return (views / 1000000).toFixed(1) + 'M';
            if (views >= 1000) return (views / 1000).toFixed(1) + 'K';
            return views;
        }
        
        function formatDate(timestamp) {
            const now = Date.now();
            const diff = now - timestamp;
            const seconds = Math.floor(diff / 1000);
            const minutes = Math.floor(seconds / 60);
            const hours = Math.floor(minutes / 60);
            const days = Math.floor(hours / 24);
            const months = Math.floor(days / 30);
            const years = Math.floor(days / 365);
            
            if (seconds < 60) return 'Just now';
            if (minutes < 60) return minutes + ' minutes ago';
            if (hours < 24) return hours + ' hours ago';
            if (days < 30) return days + ' days ago';
            if (months < 12) return months + ' months ago';
            return years + ' years ago';
        }
        
        function shuffleArray(array) {
            const shuffled = [...array];
            for (let i = shuffled.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
            }
            return shuffled;
        }
        
        function loadHome() {
            currentFilter = 'all';
            document.getElementById('searchInput').value = '';
            document.querySelectorAll('.tab').forEach((tab, i) => {
                tab.classList.toggle('active', i === 0);
            });
            renderVideos();
        }
        
        // Initialize
        loadVideos();
        
        // Close modals on background click
        document.getElementById('uploadModal').addEventListener('click', function(e) {
            if (e.target === this) closeUploadModal();
        });
        
        document.getElementById('playerModal').addEventListener('click', function(e) {
            if (e.target === this) closePlayer();
        });
    </script>
